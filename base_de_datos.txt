CREATE TABLE CLIENTES(
    ID_CLIENTES              INT              NOT NULL,
    RELA_PERSONAS            INT,
    CLIENTES_FECHA_ALTA      DATE,
    CLIENTES_ESTADO          CHAR(1)          NOT NULL,
    CLIENTES_IMPORTE_MORA    DECIMAL(6, 2)    NOT NULL,
    PRIMARY KEY (ID_CLIENTES)
)ENGINE=MYISAM
;

CREATE TABLE DETALLES(
    ID_DETALLE          INT              NOT NULL,
    RELA_PRODUCTOS      INT,
    RELA_FACTURA        INT,
    DETALLE_CANTIDAD    INT              NOT NULL,
    DETALLE_IMPORTE     DECIMAL(6, 2)    NOT NULL,
    PRIMARY KEY (ID_DETALLE)
)ENGINE=MYISAM
;

CREATE TABLE FACTURA(
    ID_FACTURAS           INT              NOT NULL,
    RELA_CLIENTES         INT,
    FACTURAS_FECHA        DATE             NOT NULL,
    FACTURAS_TIPO         CHAR(1),
    FACTURA_TOTAL_GRAL    DECIMAL(6, 2),
    PRIMARY KEY (ID_FACTURAS)
)ENGINE=MYISAM
;


CREATE TABLE PERSONAS(
    ID_PERSONAS           INT            NOT NULL,
    PERSONAS_AYN          CHAR(50),
    PERSONAS_DNI          INT,
    PERSONAS_FEC_NAC      DATE           NOT NULL,
    PERSONAS_TELEFONO     VARCHAR(20),
    PERSONAS_DOMICILIO    VARCHAR(50),
    PERSONAS_SEXO         CHAR(10),
    PRIMARY KEY (ID_PERSONAS)
)ENGINE=MYISAM
;

CREATE TABLE PRODUCTOS(
    ID_PRODUCTOS             INT              NOT NULL,
    RELA_PROVEEDORES         INT,
    PRODUCTOS_DESCRIPCION    VARCHAR(50),
    PRODUCTOS_PU             DECIMAL(6, 2),
    PRODUCTOS_STOCK          INT              NOT NULL,
    PRODUCTOS_OBS            VARCHAR(50),
    PRODUCTOS_DESCUENTO      DECIMAL(6, 2),
    PRODUCTOS_IVA            DECIMAL(6, 2)    NOT NULL,
    PRIMARY KEY (ID_PRODUCTOS)
)ENGINE=MYISAM
;


CREATE TABLE PROVEEDORES(
    ID_PROVEEDORES             INT            NOT NULL,
    PROVEEDORES_DESCRIPCION    VARCHAR(50),
    PROVEEDORES_DOMICILIO      VARCHAR(50),
    PROVEEDORES_TELEFONO       VARCHAR(20),
    PRIMARY KEY (ID_PROVEEDORES)
)ENGINE=MYISAM
;
